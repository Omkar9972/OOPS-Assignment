1.Write a standard JS function which takes a number as an argument and returns its factorial.
function factori(x) {
    let res = 1;
    for (let i = x; i > 1; i--) {
        res = res * i;

    }
    return res;
}

let y= factori(parseInt(prompt('Enter a Number >>>')));
// let y = parseInt(prompt(factori('input a number')));
